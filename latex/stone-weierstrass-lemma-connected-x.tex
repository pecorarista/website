\begin{document}
\pgfmathdeclarefunction{gauss}{2}{%
  \pgfmathparse{1/(#2*sqrt(2*pi))*exp(-((x-#1)^2)/(2*#2^2))}%
}
\begin{tikzpicture}
    \begin{axis}[
        axis x line=middle,
        axis line style=thick,
        yscale=2.2,
        xscale=1.5,
        xmin=-10,
        xmax=10,
        ymin=-1.8,

        x tick style={draw},
        x axis line style={draw},
        ytick={},
        yticklabels={},
        y axis line style={draw=none},
        y tick style={draw=none},
        samples=300,
        clip=false
    ]
    \def\SCALE{8};
    \def\EPSILON{0.05 * \SCALE};
    \def\SIGMA{3};

    % curves
    \addplot[color=sBlue, dashed][domain=-9:9] {\SCALE * gauss(-1, \SIGMA)};
    \addplot[color=sBlue, dashed][domain=-9:9] {\SCALE * gauss(0, \SIGMA)};
    \addplot[color=sBlue, dashed][domain=-9:9] {\SCALE * gauss(1, \SIGMA)};

    % curves
    \addplot[color=sRed, thick][domain=-9:{- 1 - sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}] {\SCALE * gauss(-1, \SIGMA) + \EPSILON / 2};
    \addplot[color=sRed, dashed][domain={- 1 - sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}:9] {\SCALE * gauss(-1, \SIGMA) + \EPSILON / 2};
    \addplot[color=sRed, dashed][domain=-9:9] {\SCALE * gauss(0, \SIGMA) + \EPSILON / 2};
    \addplot[color=sRed, dashed][domain=-9:{1 + sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}] {\SCALE * gauss(1, \SIGMA) + \EPSILON / 2};
    \addplot[color=sRed, thick][domain={1 + sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}:9] {\SCALE * gauss(1, \SIGMA) + \EPSILON / 2};

    % vertical lines
    \draw[thick, dotted] (axis cs: -2.5, 0) -- (axis cs: -2.5, {\SCALE / sqrt(2 * pi * \SIGMA^2) * exp(- 1.5^2 / (2 * \SIGMA^2))});
    \draw[thick, dotted] (axis cs: -1, 0) -- (axis cs: -1, {\SCALE / sqrt(2 * pi * \SIGMA^2)});
    \draw[thick, dotted] (axis cs: 0, 0) -- (axis cs: 0, {\SCALE / sqrt(2 * pi * \SIGMA^2)});
    \draw[thick, dotted] (axis cs: 1, 0) -- (axis cs: 1, {\SCALE / sqrt(2 * pi * \SIGMA^2)});
    \draw[thick, dotted] (axis cs: 2.5, 0) -- (axis cs: 2.5, {\SCALE / sqrt(2 * pi * \SIGMA^2) * exp(- 1.5^2 / (2 * \SIGMA^2))});

    \node[anchor=west] at (axis cs: 6.8, {\SCALE / sqrt(2 * pi * \SIGMA^2)}) {$1_{C(X)}$};
    \node[anchor=west] at (axis cs: 6.8, {- \EPSILON/2 + \SCALE / sqrt(2 * pi * \SIGMA^2)}) {$\displaystyle 1 - \frac{\varepsilon}{2} 1_{C(X)}$};

    \node[color=sRed] at (axis cs: -8, 1.25) {$\displaystyle \left((f_1 \lor f_2 \lor f_3) + \frac{\varepsilon}{2} 1_{C(X)}\right) \land 1_{C(X)}$};

    \draw[thick, dotted] (axis cs: -9, {\SCALE / sqrt(2 * pi * \SIGMA^2)}) -- (axis cs: 6.3, {\SCALE / sqrt(2 * pi * \SIGMA^2)});
    \draw[color=sRed, very thick] (axis cs: {- 1 - sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}, {\SCALE / sqrt(2 * pi * \SIGMA^2)}) -- (axis cs: {1 + sqrt(2) * \SIGMA * sqrt( - ln(1 - (\EPSILON * \SIGMA * sqrt(2 * pi)) / (2 * \SCALE )))}, {\SCALE / sqrt(2 * pi * \SIGMA^2)});

    % 1 - epsilon / 2
    \draw[thick, dotted] (axis cs: -9, {\SCALE / sqrt(2 * pi * \SIGMA^2) - \EPSILON / 2}) -- (axis cs: 6.3, {\SCALE / sqrt(2 * pi * \SIGMA^2) - \EPSILON / 2});

    % epsilon

    % epsilon / 2

    \draw[thick, dotted] (axis cs: -9, {\EPSILON / 2}) -- (axis cs: 9, {\EPSILON / 2});

    % intervals
    \node at (axis cs: 3, -0.18) {$K$};
    \draw[ultra thick] (axis cs: -2.5, 0) -- (axis cs: 2.5, 0);

    \node at (axis cs: 8, -0.18) {$X \setminus U$};
    \draw[ultra thick, color=sGreen] (axis cs: 7, 0) -- (axis cs: 9, 0);
    \draw[ultra thick, color=sGreen] (axis cs: -9, 0) -- (axis cs: -7, 0);

    \end{axis}
    \end{tikzpicture}
