    \begin{broadalgo}
        \Procedure{Insertion-Sort}{$A$}
        \ForTo{$j = 2$}{$A.\mathrm{length}$}
            \State{$k = A[j]$}
                \State{$i = j - 1$}
                \While{$i > 0$ and $A[i] > k$}
                    \State{$A[i + 1] = A[i]$}
                    \State{$i = i - 1$}
                \EndWhile
                \State{$A[i + 1] = k$}
            \EndFor
        \EndProcedure
    \end{broadalgo}
